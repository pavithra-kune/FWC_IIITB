\end{itemize}

If canvas costs Rs.120 per sq.m, find the amount shared by each school.

\textbf{Write the following functions in the simplest form:}

\begin{enumerate}

\item[5.]

\begin{align*}
\tan^{-1}
\left(
\frac{\sqrt{1+x^2}-1}{x}
\right),
\quad x \neq 0
\end{align*}

\item[6.]

\begin{align*}
\tan^{-1}
\left(
\frac{1}{\sqrt{x^2-1}}
\right),
\quad |x|>1
\end{align*}

\item[7.]

\begin{align*}
\tan^{-1}
\left(
\sqrt{
\frac{1-\cos x}{1+\cos x}
}
\right),
\quad 0<x<\pi
\end{align*}

\item[8.]

\begin{align*}
\tan^{-1}
\left(
\frac{\cos x-\sin x}
{\cos x+\sin x}
\right),
\quad
-\frac{\pi}{4}
<
x
<
\frac{3\pi}{4}
\end{align*}

\item[9.]

\begin{align*}
\tan^{-1}
\left(
\frac{x}{\sqrt{a^2-x^2}}
\right),
\quad |x|<a
\end{align*}

\item[10.]

\begin{align*}
\tan^{-1}
\left(
\frac{3a^2x-x^3}
{a^3-3ax^2}
\right)
\end{align*}

\begin{align*}
a>0,
\quad
-\frac{a}{\sqrt{3}}
<
x
<
\frac{a}{\sqrt{3}}
\end{align*}

\end{enumerate}
